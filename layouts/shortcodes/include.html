{{/* This will try to find a resource in the "includes" bundle */}} {{- $name := .Get 0 -}} {{- if $name -}} {{- $bundle := site.GetPage "page" "includes" -}} {{- with $bundle -}} {{- $pattern := printf "%s*" $name -}} {{- range $bundle.Resources -}} {{- end -}} {{- $resource := $bundle.Resources.GetMatch $pattern -}} {{- with $resource -}} {{- .Content | safeHTML -}} {{- else -}} {{/* It is not a resource in the includes bundle. Try to find the page via content path. */}} {{- $path := strings.TrimSuffix ".md" (path.Join $.Page.File.Dir $name) -}} {{- $page := site.GetPage "page" $path -}} {{- if not $page -}} {{- $page = site.GetPage "page" (path.Join "/" $path) -}} {{- end -}} {{- with $page }} {{ .Content }} {{- else -}} {{ warnf "[%s] no Resource or Page matching %q." $.Page.Lang ($name) }} {{- end -}} {{- end -}} {{- else -}} {{ warnf "[%s] the 'includes' bundle was not found." $.Page.Lang }} {{- end -}} {{- else -}} {{- errorf "[%s] missing resource name in include for page %q" $.Page.Lang $.Page.Path -}} {{- end -}}
